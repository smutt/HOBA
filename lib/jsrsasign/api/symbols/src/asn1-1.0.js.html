 84 KJUR.asn1.ASN1Util = new function() {
 85     this.integerToByteHex = function(i) {
 86 	var h = i.toString(16);
 87 	if (h.length == 1) h = '0' + h;
 88 	return h;
 89     };
 90     this.bigIntToMinTwosComplementsHex = function(bigIntegerValue) {
 91 	var h = bigIntegerValue.toString(16);
 92 	if (h.substr(0, 1) != '-') {
 93 	    if (h.length % 2 == 1) {
 94 		h = '0' + h;
 95 	    } else {
 96 		if (! h.match(/^[0-7]/)) {
 97 		    h = '00' + h;
 98 		}
 99 	    }
100 	} else {
101 	    var hPos = h.substr(1);
102 	    var xorLen = hPos.length;
103 	    if (xorLen % 2 == 1) {
104 		xorLen += 1;
105 	    } else {
106 		if (! h.match(/^[0-7]/)) {
107 		    xorLen += 2;
108 		}
109 	    }
110 	    var hMask = '';
111 	    for (var i = 0; i < xorLen; i++) {
112 		hMask += 'f';
113 	    }
114 	    var biMask = new BigInteger(hMask, 16);
115 	    var biNeg = biMask.xor(bigIntegerValue).add(BigInteger.ONE);
116 	    h = biNeg.toString(16).replace(/^-/, '');
117 	}
118 	return h;
119     };
120 };

128 /**
129  * base class for ASN.1 DER encoder object
130  * @name KJUR.asn1.ASN1Object
131  * @class base class for ASN.1 DER encoder object
132  * @property {Boolean} isModified flag whether internal data was changed
133  * @property {String} hTLV hexadecimal string of ASN.1 TLV
134  * @property {String} hT hexadecimal string of ASN.1 TLV tag(T)
135  * @property {String} hL hexadecimal string of ASN.1 TLV length(L)
136  * @property {String} hV hexadecimal string of ASN.1 TLV value(V)
137  * @description
138  */
139 KJUR.asn1.ASN1Object = function() {
140     var isModified = true;
141     var hTLV = null;
142     var hT = '00'
143     var hL = '00';
144     var hV = '';
145 
146     /**
147      * get hexadecimal ASN.1 TLV length(L) bytes from TLV value(V)
148      * @name getLengthHexFromValue
149      * @memberOf KJUR.asn1.ASN1Object
150      * @function
151      * @return {String} hexadecimal string of ASN.1 TLV length(L)
152      */
153     this.getLengthHexFromValue = function() {
154 	if (typeof this.hV == "undefined" || this.hV == null) {
155 	    throw "this.hV is null or undefined.";
156 	}
157 	if (this.hV.length % 2 == 1) {
158 	    throw "value hex must be even length: n=" + hV.length + ",v=" + this.hV;
159 	}
160 	var n = this.hV.length / 2;
161 	var hN = n.toString(16);
162 	if (hN.length % 2 == 1) {
163 	    hN = "0" + hN;
164 	}
165 	if (n < 128) {
166 	    return hN;
167 	} else {
168 	    var hNlen = hN.length / 2;
169 	    if (hNlen > 15) {
170 		throw "ASN.1 length too long to represent by 8x: n = " + n.toString(16);
171 	    }
172 	    var head = 128 + hNlen;
173 	    return head.toString(16) + hN;
174 	}
175     };
176 
177     /**
178      * get hexadecimal string of ASN.1 TLV bytes
179      * @name getEncodedHex
180      * @memberOf KJUR.asn1.ASN1Object
181      * @function
182      * @return {String} hexadecimal string of ASN.1 TLV
183      */
184     this.getEncodedHex = function() {
185 	if (this.hTLV == null || this.isModified) {
186 	    this.hV = this.getFreshValueHex();
187 	    this.hL = this.getLengthHexFromValue();
188 	    this.hTLV = this.hT + this.hL + this.hV;
189 	    this.isModified = false;
190 	    //alert("first time: " + this.hTLV);
191 	}
192 	return this.hTLV;
193     };
194 
195     /**
196      * get hexadecimal string of ASN.1 TLV value(V) bytes
197      * @name getValueHex
198      * @memberOf KJUR.asn1.ASN1Object
199      * @function
200      * @return {String} hexadecimal string of ASN.1 TLV value(V) bytes
201      */
202     this.getValueHex = function() {
203 	this.getEncodedHex();
204 	return this.hV;
205     }
206 
207     this.getFreshValueHex = function() {
208 	return '';
209     };
210 };

290 /**
291  * base class for ASN.1 DER Generalized/UTCTime class
292  * @name KJUR.asn1.DERAbstractTime
293  * @class base class for ASN.1 DER Generalized/UTCTime class
294  * @param {Array} params associative array of parameters (ex. {'str': '130430235959Z'})
295  * @extends KJUR.asn1.ASN1Object
296  * @description
297  * @see KJUR.asn1.ASN1Object - superclass
298  */
299 KJUR.asn1.DERAbstractTime = function(params) {
300     KJUR.asn1.DERAbstractTime.superclass.constructor.call(this);
301     var s = null;
302     var date = null;
303 
304     // --- PRIVATE METHODS --------------------
305     this.localDateToUTC = function(d) {
306 	utc = d.getTime() + (d.getTimezoneOffset() * 60000);
307 	var utcDate = new Date(utc);
308 	return utcDate;
309     };
310 
311     this.formatDate = function(dateObject, type) {
312 	var pad = this.zeroPadding;
313 	var d = this.localDateToUTC(dateObject);
314 	var year = String(d.getFullYear());
315 	if (type == 'utc') year = year.substr(2, 2);
316 	var month = pad(String(d.getMonth() + 1), 2);
317 	var day = pad(String(d.getDate()), 2);
318 	var hour = pad(String(d.getHours()), 2);
319 	var min = pad(String(d.getMinutes()), 2);
320 	var sec = pad(String(d.getSeconds()), 2);
321 	return year + month + day + hour + min + sec + 'Z';
322     };
323 
324     this.zeroPadding = function(s, len) {
325 	if (s.length >= len) return s;
326 	return new Array(len - s.length + 1).join('0') + s;
327     };
328 
329     // --- PUBLIC METHODS --------------------
330     /**
331      * get string value of this string object
332      * @name getString
333      * @memberOf KJUR.asn1.DERAbstractTime
334      * @function
335      * @return {String} string value of this time object
336      */
337     this.getString = function() {
338 	return this.s;
339     };
340 
341     /**
342      * set value by a string
343      * @name setString
344      * @memberOf KJUR.asn1.DERAbstractTime
345      * @function
346      * @param {String} newS value by a string to set such like "130430235959Z"
347      */
348     this.setString = function(newS) {
349 	this.hTLV = null;
350 	this.isModified = true;
351 	this.s = newS;
352 	this.hV = stohex(this.s);
353     };
354 
355     /**
356      * set value by a Date object
357      * @name setByDateValue
358      * @memberOf KJUR.asn1.DERAbstractTime
359      * @function
360      * @param {Integer} year year of date (ex. 2013)
361      * @param {Integer} month month of date between 1 and 12 (ex. 12)
362      * @param {Integer} day day of month
363      * @param {Integer} hour hours of date
364      * @param {Integer} min minutes of date
365      * @param {Integer} sec seconds of date
366      */
367     this.setByDateValue = function(year, month, day, hour, min, sec) {
368 	var dateObject = new Date(Date.UTC(year, month - 1, day, hour, min, sec, 0));
369 	this.setByDate(dateObject);
370     };
371 
372     this.getFreshValueHex = function() {
373 	return this.hV;
374     };
375 };

527 /**
528  * class for ASN.1 DER encoded BitString primitive
529  * @name KJUR.asn1.DERBitString
530  * @class class for ASN.1 DER encoded BitString primitive
531  * @extends KJUR.asn1.ASN1Object
532  * @description 
533  * <br/>
534  * As for argument 'params' for constructor, you can specify one of
535  * following properties:
536  * <ul>
537  * <li>bin - specify binary string (ex. '10111')</li>
538  * <li>array - specify array of boolean (ex. [true,false,true,true])</li>
539  * <li>hex - specify hexadecimal string of ASN.1 value(V) including unused bits</li>
540  * </ul>
541  * NOTE: 'params' can be omitted.
542  */
543 KJUR.asn1.DERBitString = function(params) {
544     KJUR.asn1.DERBitString.superclass.constructor.call(this);
545     this.hT = "03";
546 
547     /**
548      * set ASN.1 value(V) by a hexadecimal string including unused bits
549      * @name setHexValueIncludingUnusedBits
550      * @memberOf KJUR.asn1.DERBitString
551      * @function
552      * @param {String} newHexStringIncludingUnusedBits
553      */
554     this.setHexValueIncludingUnusedBits = function(newHexStringIncludingUnusedBits) {
555 	this.hTLV = null;
556 	this.isModified = true;
557 	this.hV = newHexStringIncludingUnusedBits;
558     };
559 
560     /**
561      * set ASN.1 value(V) by unused bit and hexadecimal string of value
562      * @name setUnusedBitsAndHexValue
563      * @memberOf KJUR.asn1.DERBitString
564      * @function
565      * @param {Integer} unusedBits
566      * @param {String} hValue
567      */
568     this.setUnusedBitsAndHexValue = function(unusedBits, hValue) {
569 	if (unusedBits < 0 || 7 < unusedBits) {
570 	    throw "unused bits shall be from 0 to 7: u = " + unusedBits;
571 	}
572 	var hUnusedBits = "0" + unusedBits;
573 	this.hTLV = null;
574 	this.isModified = true;
575 	this.hV = hUnusedBits + hValue;
576     };
577 
578     /**
579      * set ASN.1 DER BitString by binary string
580      * @name setByBinaryString
581      * @memberOf KJUR.asn1.DERBitString
582      * @function
583      * @param {String} binaryString binary value string (i.e. '10111')
584      * @description
585      * Its unused bits will be calculated automatically by length of 
586      * 'binaryValue'. <br/>
587      * NOTE: Trailing zeros '0' will be ignored.
588      */
589     this.setByBinaryString = function(binaryString) {
590 	binaryString = binaryString.replace(/0+$/, '');
591 	var unusedBits = 8 - binaryString.length % 8;
592 	if (unusedBits == 8) unusedBits = 0;
593 	for (var i = 0; i <= unusedBits; i++) {
594 	    binaryString += '0';
595 	}
596 	var h = '';
597 	for (var i = 0; i < binaryString.length - 1; i += 8) {
598 	    var b = binaryString.substr(i, 8);
599 	    var x = parseInt(b, 2).toString(16);
600 	    if (x.length == 1) x = '0' + x;
601 	    h += x;  
602 	}
603 	this.hTLV = null;
604 	this.isModified = true;
605 	this.hV = '0' + unusedBits + h;
606     };
607 
608     /**
609      * set ASN.1 TLV value(V) by an array of boolean
610      * @name setByBooleanArray
611      * @memberOf KJUR.asn1.DERBitString
612      * @function
613      * @param {array} booleanArray array of boolean (ex. [true, false, true])
614      * @description
615      * NOTE: Trailing falses will be ignored.
