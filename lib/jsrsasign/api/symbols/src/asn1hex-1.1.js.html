 39 /**
 40  * get byte length for ASN.1 L(length) bytes
 41  * @name getByteLengthOfL_AtObj
 42  * @memberOf ASN1HEX
 43  * @function
 44  * @param {String} s hexadecimal string of ASN.1 DER encoded data
 45  * @param {Number} pos string index
 46  * @return byte length for ASN.1 L(length) bytes
 47  */
 48 function _asnhex_getByteLengthOfL_AtObj(s, pos) {
 49   if (s.substring(pos + 2, pos + 3) != '8') return 1;
 50   var i = parseInt(s.substring(pos + 3, pos + 4));
 51   if (i == 0) return -1; 		// length octet '80' indefinite length
 52   if (0 < i && i < 10) return i + 1;	// including '8?' octet;
 53   return -2;				// malformed format
 54 }

 57 /**
 58  * get hexadecimal string for ASN.1 L(length) bytes
 59  * @name getHexOfL_AtObj
 60  * @memberOf ASN1HEX
 61  * @function
 62  * @param {String} s hexadecimal string of ASN.1 DER encoded data
 63  * @param {Number} pos string index
 64  * @return {String} hexadecimal string for ASN.1 L(length) bytes
 65  */
 66 function _asnhex_getHexOfL_AtObj(s, pos) {
 67   var len = _asnhex_getByteLengthOfL_AtObj(s, pos);
 68   if (len < 1) return '';
 69   return s.substring(pos + 2, pos + 2 + len * 2);
 70 }

 72 //
 73 //   getting ASN.1 length value at the position 'idx' of
 74 //   hexa decimal string 's'.
 75 //
 76 //   f('3082025b02...', 0) ... 82025b ... ???
 77 //   f('020100', 0) ... 01 ... 1
 78 //   f('0203001...', 0) ... 03 ... 3
 79 //   f('02818003...', 0) ... 8180 ... 128
 80 /**
 81  * get integer value of ASN.1 length for ASN.1 data
 82  * @name getIntOfL_AtObj
 83  * @memberOf ASN1HEX
 84  * @function
 85  * @param {String} s hexadecimal string of ASN.1 DER encoded data
 86  * @param {Number} pos string index
 87  * @return ASN.1 L(length) integer value
 88  */
 89 function _asnhex_getIntOfL_AtObj(s, pos) {
 90   var hLength = _asnhex_getHexOfL_AtObj(s, pos);
 91   if (hLength == '') return -1;
 92   var bi;
 93   if (parseInt(hLength.substring(0, 1)) < 8) {
 94      bi = parseBigInt(hLength, 16);
 95   } else {
 96      bi = parseBigInt(hLength.substring(2), 16);
 97   }
 98   return bi.intValue();
 99 }

101 /**
102  * get ASN.1 value starting string position for ASN.1 object refered by index 'idx'.
103  * @name getStartPosOfV_AtObj
104  * @memberOf ASN1HEX
105  * @function
106  * @param {String} s hexadecimal string of ASN.1 DER encoded data
107  * @param {Number} pos string index
108  */
109 function _asnhex_getStartPosOfV_AtObj(s, pos) {
110   var l_len = _asnhex_getByteLengthOfL_AtObj(s, pos);
111   if (l_len < 0) return l_len;
112   return pos + (l_len + 1) * 2;
113 }

115 /**
116  * get hexadecimal string of ASN.1 V(value)
117  * @name getHexOfV_AtObj
118  * @memberOf ASN1HEX
119  * @function
120  * @param {String} s hexadecimal string of ASN.1 DER encoded data
121  * @param {Number} pos string index
122  * @return {String} hexadecimal string of ASN.1 value.
123  */
124 function _asnhex_getHexOfV_AtObj(s, pos) {
125   var pos1 = _asnhex_getStartPosOfV_AtObj(s, pos);
126   var len = _asnhex_getIntOfL_AtObj(s, pos);
127   return s.substring(pos1, pos1 + len * 2);
128 }

130 /**
131  * get hexadecimal string of ASN.1 TLV at
132  * @name getHexOfTLV_AtObj
133  * @memberOf ASN1HEX
134  * @function
135  * @param {String} s hexadecimal string of ASN.1 DER encoded data
136  * @param {Number} pos string index
137  * @return {String} hexadecimal string of ASN.1 TLV.
138  * @since 1.1
139  */
140 function _asnhex_getHexOfTLV_AtObj(s, pos) {
141   var hT = s.substr(pos, 2);
142   var hL = _asnhex_getHexOfL_AtObj(s, pos);
143   var hV = _asnhex_getHexOfV_AtObj(s, pos);
144   return hT + hL + hV;
145 }

147 /**
148  * get next sibling starting index for ASN.1 object string
149  * @name getPosOfNextSibling_AtObj
150  * @memberOf ASN1HEX
151  * @function
152  * @param {String} s hexadecimal string of ASN.1 DER encoded data
153  * @param {Number} pos string index
154  * @return next sibling starting index for ASN.1 object string
155  */
156 function _asnhex_getPosOfNextSibling_AtObj(s, pos) {
157   var pos1 = _asnhex_getStartPosOfV_AtObj(s, pos);
158   var len = _asnhex_getIntOfL_AtObj(s, pos);
159   return pos1 + len * 2;
160 }

162 /**
163  * get array of indexes of child ASN.1 objects
164  * @name getPosArrayOfChildren_AtObj
165  * @memberOf ASN1HEX
166  * @function
167  * @param {String} s hexadecimal string of ASN.1 DER encoded data
168  * @param {Number} start string index of ASN.1 object
169  * @return {Array of Number} array of indexes for childen of ASN.1 objects
170  */
171 function _asnhex_getPosArrayOfChildren_AtObj(h, pos) {
172   var a = new Array();
173   var p0 = _asnhex_getStartPosOfV_AtObj(h, pos);
174   a.push(p0);
175 
176   var len = _asnhex_getIntOfL_AtObj(h, pos);
177   var p = p0;
178   var k = 0;
179   while (1) {
180     var pNext = _asnhex_getPosOfNextSibling_AtObj(h, p);
181     if (pNext == null || (pNext - p0  >= (len * 2))) break;
182     if (k >= 200) break;
183 
184     a.push(pNext);
185     p = pNext;
186 
187     k++;
188   }
189 
190   return a;
191 }

193 /**
194  * get string index of nth child object of ASN.1 object refered by h, idx
195  * @name getNthChildIndex_AtObj
196  * @memberOf ASN1HEX
197  * @function
198  * @param {String} h hexadecimal string of ASN.1 DER encoded data
199  * @param {Number} idx start string index of ASN.1 object
200  * @param {Number} nth for child
201  * @return {Number} string index of nth child.
202  * @since 1.1
203  */
204 function _asnhex_getNthChildIndex_AtObj(h, idx, nth) {
205   var a = _asnhex_getPosArrayOfChildren_AtObj(h, idx);
206   return a[nth];
207 }

211 /**
212  * get string index of nth child object of ASN.1 object refered by h, idx
213  * @name getDecendantIndexByNthList
214  * @memberOf ASN1HEX
215  * @function
216  * @param {String} h hexadecimal string of ASN.1 DER encoded data
217  * @param {Number} currentIndex start string index of ASN.1 object
218  * @param {Array of Number} nthList array list of nth
219  * @return {Number} string index refered by nthList
220  * @since 1.1
221  */
222 function _asnhex_getDecendantIndexByNthList(h, currentIndex, nthList) {
223   if (nthList.length == 0) {
224     return currentIndex;
225   }
226   var firstNth = nthList.shift();
227   var a = _asnhex_getPosArrayOfChildren_AtObj(h, currentIndex);
228   return _asnhex_getDecendantIndexByNthList(h, a[firstNth], nthList);
229 }

242 function _asnhex_getDecendantHexTLVByNthList(h, currentIndex, nthList) {

244   return _asnhex_getHexOfTLV_AtObj(h, idx);
