 26 /**
 27  * @name PKCS5PKEY
 28  * @class class for PKCS#5 and PKCS#8 private key 
 29  * @description 
 30  * <br/>
 31  * {@link PKCS5PKEY} class has following features:
 32  * <ul>
 33  * <li>read and parse PEM formatted encrypted PKCS#5 private key
 34  * <li>generate PEM formatted encrypted PKCS#5 private key
 35  * <li>read and parse PEM formatted plain PKCS#8 private key
 36  * </ul>
 37  * Currently supports only RSA private key and
 38  * following symmetric key algorithms to protect private key.
 39  * <ul>
 40  * <li>DES-EDE3-CBC</li>
 41  * <li>AES-256-CBC</li>
 42  * <li>AES-192-CBC</li>
 43  * <li>AES-128-CBC</li>
 44  * </ul>
 45  * 
 46  * @example
 47  * Here is an example of PEM formatted encrypted PKCS#5 private key.
 48  * -----BEGIN RSA PRIVATE KEY-----
 49  * Proc-Type: 4,ENCRYPTED
 50  * DEK-Info: AES-256-CBC,40555967F759530864FE022E257DE34E
 51  *
 52  * jV7uXajRw4cccDaliagcqiLOiQEUCe19l761pXRxzgQP+DH4rCi12T4puTdZyy6l
 53  *          ...(snip)...
 54  * qxLS+BASmyGm4DME6m+kltZ12LXwPgNU6+d+XQ4NXSA=
 55  *-----END RSA PRIVATE KEY-----
 56  */

 62     // shared key decryption ------------------------------------------
 63     var decryptAES = function(dataHex, keyHex, ivHex) {
 64 	return decryptGeneral(CryptoJS.AES, dataHex, keyHex, ivHex);
 65     };
 66 
 67     var decrypt3DES = function(dataHex, keyHex, ivHex) {
 68 	return decryptGeneral(CryptoJS.TripleDES, dataHex, keyHex, ivHex);
 69     };

 71     var decryptGeneral = function(f, dataHex, keyHex, ivHex) {
 72 	var data = CryptoJS.enc.Hex.parse(dataHex);
 73 	var key = CryptoJS.enc.Hex.parse(keyHex);
 74 	var iv = CryptoJS.enc.Hex.parse(ivHex);
 75 	var encrypted = {};
 76 	encrypted.key = key;
 77 	encrypted.iv = iv;
 78 	encrypted.ciphertext = data;
 79 	var decrypted = f.decrypt(encrypted, key, { iv: iv });
 80 	return CryptoJS.enc.Hex.stringify(decrypted);
 81     };
 82 
 83     // shared key decryption ------------------------------------------
 84     var encryptAES = function(dataHex, keyHex, ivHex) {
 85 	return encryptGeneral(CryptoJS.AES, dataHex, keyHex, ivHex);
 86     };
 87 
 88     var encrypt3DES = function(dataHex, keyHex, ivHex) {
 89 	return encryptGeneral(CryptoJS.TripleDES, dataHex, keyHex, ivHex);
 90     };

 92     var encryptGeneral = function(f, dataHex, keyHex, ivHex) {
 93 	var data = CryptoJS.enc.Hex.parse(dataHex);
 94 	var key = CryptoJS.enc.Hex.parse(keyHex);
 95 	var iv = CryptoJS.enc.Hex.parse(ivHex);
 96 	var msg = {};
 97 	var encryptedHex = f.encrypt(data, key, { iv: iv });
 98         var encryptedWA = CryptoJS.enc.Hex.parse(encryptedHex.toString());
 99         var encryptedB64 = CryptoJS.enc.Base64.stringify(encryptedWA);
100 	return encryptedB64;
101     };

103     // other methods --------------------------------------------------

105 	return ALGLIST[algName]['proc'];

108     var _generateIvSaltHex = function(numBytes) {
109 	var wa = CryptoJS.lib.WordArray.random(numBytes);
110 	var hex = CryptoJS.enc.Hex.stringify(wa);
111 	return hex;
112     };

114     var ALGLIST = {
115 	'AES-256-CBC': { 'proc': decryptAES, 'eproc': encryptAES, keylen: 32, ivlen: 16 },
116 	'AES-192-CBC': { 'proc': decryptAES, 'eproc': encryptAES, keylen: 24, ivlen: 16 },
117 	'AES-128-CBC': { 'proc': decryptAES, 'eproc': encryptAES, keylen: 16, ivlen: 16 },
118 	'DES-EDE3-CBC': { 'proc': decrypt3DES, 'eproc': encrypt3DES, keylen: 24, ivlen: 8 }

121     var _parsePKCS5PEM = function(sPKCS5PEM) {
122 	var info = {};
123 	if (sPKCS5PEM.match(new RegExp("DEK-Info: ([^,]+),([0-9A-Fa-f]+)", "m"))) {
124 	    info.cipher = RegExp.$1;
125 	    info.ivsalt = RegExp.$2;
126 	}
127 	if (sPKCS5PEM.match(new RegExp("-----BEGIN ([A-Z]+) PRIVATE KEY-----"))) {
128 	    info.type = RegExp.$1;
129 	}
130 	var i1 = -1;
131 	var lenNEWLINE = 0;
132 	if (sPKCS5PEM.indexOf("\r\n\r\n") != -1) {
133 	    i1 = sPKCS5PEM.indexOf("\r\n\r\n");
134 	    lenNEWLINE = 2;
135 	}
136 	if (sPKCS5PEM.indexOf("\n\n") != -1) {
137 	    i1 = sPKCS5PEM.indexOf("\n\n");
138 	    lenNEWLINE = 1;
139 	}
140 	var i2 = sPKCS5PEM.indexOf("-----END");
141 	if (i1 != -1 && i2 != -1) {
142 	    var s = sPKCS5PEM.substring(i1 + lenNEWLINE * 2, i2 - lenNEWLINE);
143 	    s = s.replace(/\s+/g, '');
144 	    info.data = s;
145 	}
146 	return info;
147     };

149     var _getKeyAndUnusedIvByPasscodeAndIvsalt = function(algName, passcode, ivsaltHex) {
150 	//alert("ivsaltHex(2) = " + ivsaltHex);
151 	var saltHex = ivsaltHex.substring(0, 16);
152 	//alert("salt = " + saltHex);
153 	    
154 	var salt = CryptoJS.enc.Hex.parse(saltHex);
155 	var data = CryptoJS.enc.Utf8.parse(passcode);
156 	//alert("salt = " + salt);
157 	//alert("data = " + data);
158 
159 	var nRequiredBytes = ALGLIST[algName]['keylen'] + ALGLIST[algName]['ivlen'];
160 	var hHexValueJoined = '';
161 	var hLastValue = null;
162 	//alert("nRequiredBytes = " + nRequiredBytes);
163 	for (;;) {
164 	    var h = CryptoJS.algo.MD5.create();
165 	    if (hLastValue != null) {
166 		h.update(hLastValue);
167 	    }
168 	    h.update(data);
169 	    h.update(salt);
170 	    hLastValue = h.finalize();
171 	    hHexValueJoined = hHexValueJoined + CryptoJS.enc.Hex.stringify(hLastValue);
172 	    //alert("joined = " + hHexValueJoined);
173 	    if (hHexValueJoined.length >= nRequiredBytes * 2) {
174 		break;
175 	    }
176 	}
177 	var result = {};
178 	result.keyhex = hHexValueJoined.substr(0, ALGLIST[algName]['keylen'] * 2);
179 	result.ivhex = hHexValueJoined.substr(ALGLIST[algName]['keylen'] * 2, ALGLIST[algName]['ivlen'] * 2);
180 	return result;
181     };

183     /*
184      * @param {String} privateKeyB64 base64 string of encrypted private key
185      * @param {String} sharedKeyAlgName algorithm name of shared key encryption
186      * @param {String} sharedKeyHex hexadecimal string of shared key to encrypt
187      * @param {String} ivsaltHex hexadecimal string of IV and salt
188      * @param {String} hexadecimal string of decrypted private key
189      */
190     var _decryptKeyB64 = function(privateKeyB64, sharedKeyAlgName, sharedKeyHex, ivsaltHex) {
191 	var privateKeyWA = CryptoJS.enc.Base64.parse(privateKeyB64);
192 	var privateKeyHex = CryptoJS.enc.Hex.stringify(privateKeyWA);
193 	var f = ALGLIST[sharedKeyAlgName]['proc'];
194 	var decryptedKeyHex = f(privateKeyHex, sharedKeyHex, ivsaltHex);
195 	return decryptedKeyHex;
196     };

202      * @param {String} ivsaltHex hexadecimal string of IV and salt
203      * @param {String} base64 string of encrypted private key
204      */
205     var _encryptKeyHex = function(privateKeyHex, sharedKeyAlgName, sharedKeyHex, ivsaltHex) {
206 	var f = ALGLIST[sharedKeyAlgName]['eproc'];
207 	var encryptedKeyB64 = f(privateKeyHex, sharedKeyHex, ivsaltHex);
208 	return encryptedKeyB64;
209     };

275 	/**
276          * decrypt PEM formatted protected PKCS#5 private key with passcode
277 	 * @name getDecryptedKeyHex
278 	 * @memberOf PKCS5PKEY
279 	 * @function
280 	 * @param {String} sEncryptedPEM PEM formatted protected passcode protected PKCS#5 private key
281 	 * @param {String} passcode passcode to decrypt private key (ex. 'password')
282 	 * @return {String} hexadecimal string of decrypted RSA priavte key
283 	 */
284 	getDecryptedKeyHex: function(sEncryptedPEM, passcode) {
285 	    // 1. parse pem
286 	    var info = _parsePKCS5PEM(sEncryptedPEM);
287 	    var publicKeyAlgName = info.type;
288 	    var sharedKeyAlgName = info.cipher;
289 	    var ivsaltHex = info.ivsalt;
290 	    var privateKeyB64 = info.data;
291 	    //alert("ivsaltHex = " + ivsaltHex);
292 
293 	    // 2. generate shared key
294 	    var sharedKeyInfo = _getKeyAndUnusedIvByPasscodeAndIvsalt(sharedKeyAlgName, passcode, ivsaltHex);
295 	    var sharedKeyHex = sharedKeyInfo.keyhex;
296 	    //alert("sharedKeyHex = " + sharedKeyHex);
297 
298 	    // 3. decrypt private key
299             var decryptedKey = _decryptKeyB64(privateKeyB64, sharedKeyAlgName, sharedKeyHex, ivsaltHex);
300 	    return decryptedKey;

319 	/**
320          * get PEM formatted encrypted PKCS#5 private key from hexadecimal string of plain private key
321 	 * @name getEryptedPKCS5PEMFromPrvKeyHex
322 	 * @memberOf PKCS5PKEY
323 	 * @function
324 	 * @param {String} hPrvKey hexadecimal string of plain private key
325 	 * @param {String} passcode pass code to protect private key (ex. password)
326 	 * @param {String} sharedKeyAlgName algorithm name to protect private key (ex. AES-256-CBC)
327 	 * @param {String} ivsaltHex hexadecimal string of IV and salt
328 	 * @return {String} string of PEM formatted encrypted PKCS#5 private key
329          * @since pkcs5pkey 1.0.2
330 	 * @description
331 	 * <br/>
332 	 * generate PEM formatted encrypted PKCS#5 private key by hexadecimal string encoded
333 	 * ASN.1 object of plain RSA private key.
334 	 * Following arguments can be omitted.
335 	 * <ul>
336 	 * <li>alg - AES-256-CBC will be used if omitted.</li>
337 	 * <li>ivsaltHex - automatically generate IV and salt which length depends on algorithm</li>
338 	 * </ul>
339 	 * @example
340 	 * var pem = 
341          *   PKCS5PKEY.getEryptedPKCS5PEMFromPrvKeyHex(plainKeyHex, "password");
342 	 * var pem2 = 
343          *   PKCS5PKEY.getEryptedPKCS5PEMFromPrvKeyHex(plainKeyHex, "password", "AES-128-CBC");
344 	 * var pem3 = 
345          *   PKCS5PKEY.getEryptedPKCS5PEMFromPrvKeyHex(plainKeyHex, "password", "AES-128-CBC", "1f3d02...");
346 	 */
347 	getEryptedPKCS5PEMFromPrvKeyHex: function(hPrvKey, passcode, sharedKeyAlgName, ivsaltHex) {
348 	    var sPEM = "";
349 
350 	    // 1. set sharedKeyAlgName if undefined (default AES-256-CBC)
351 	    if (typeof sharedKeyAlgName == "undefined" || sharedKeyAlgName == null) {
352 		sharedKeyAlgName = "AES-256-CBC";
353 	    }
354 	    if (typeof ALGLIST[sharedKeyAlgName] == "undefined")
355 		throw "PKCS5PKEY unsupported algorithm: " + sharedKeyAlgName;
356 
357 	    // 2. set ivsaltHex if undefined
358 	    if (typeof ivsaltHex == "undefined" || ivsaltHex == null) {
359 		var ivlen = ALGLIST[sharedKeyAlgName]['ivlen'];
360 		var randIV = _generateIvSaltHex(ivlen);
361 		ivsaltHex = randIV.toUpperCase();
362 	    }
363 
364 	    // 3. get shared key
365             //alert("ivsalthex=" + ivsaltHex);
366 	    var sharedKeyInfo = _getKeyAndUnusedIvByPasscodeAndIvsalt(sharedKeyAlgName, passcode, ivsaltHex);
367 	    var sharedKeyHex = sharedKeyInfo.keyhex;
368 	    // alert("sharedKeyHex = " + sharedKeyHex);
369 
370             // 3. get encrypted Key in Base64
371             var encryptedKeyB64 = _encryptKeyHex(hPrvKey, sharedKeyAlgName, sharedKeyHex, ivsaltHex);
372 
373 	    var pemBody = encryptedKeyB64.replace(/(.{64})/g, "$1\r\n");
374 	    var sPEM = "-----BEGIN RSA PRIVATE KEY-----\r\n";
375 	    sPEM += "Proc-Type: 4,ENCRYPTED\r\n";
376 	    sPEM += "DEK-Info: " + sharedKeyAlgName + "," + ivsaltHex + "\r\n";
377 	    sPEM += "\r\n";
378 	    sPEM += pemBody;
379 	    sPEM += "\r\n-----END RSA PRIVATE KEY-----\r\n";
380 
381 	    return sPEM;
382         },

384 	/**
385          * get PEM formatted encrypted PKCS#5 private key from RSAKey object of private key
386 	 * @name getEryptedPKCS5PEMFromRSAKey
387 	 * @memberOf PKCS5PKEY
388 	 * @function
389 	 * @param {RSAKey} pKey RSAKey object of private key
390 	 * @param {String} passcode pass code to protect private key (ex. password)
391 	 * @param {String} alg algorithm name to protect private key (default AES-256-CBC)
392 	 * @param {String} ivsaltHex hexadecimal string of IV and salt (default generated random IV)
393 	 * @return {String} string of PEM formatted encrypted PKCS#5 private key
394          * @since pkcs5pkey 1.0.2
395 	 * @description
396 	 * <br/>
397 	 * generate PEM formatted encrypted PKCS#5 private key by
398 	 * {@link RSAKey} object of RSA private key and passcode.
399 	 * Following argument can be omitted.
400 	 * <ul>
401 	 * <li>alg - AES-256-CBC will be used if omitted.</li>
402 	 * <li>ivsaltHex - automatically generate IV and salt which length depends on algorithm</li>
403 	 * </ul>
404 	 * @example
405 	 * var pkey = new RSAKey();
406 	 * pkey.generate(1024, '10001'); // generate 1024bit RSA private key with public exponent 'x010001'
407 	 * var pem = PKCS5PKEY.getEryptedPKCS5PEMFromRSAKey(pkey, "password");
408 	 */
409         getEryptedPKCS5PEMFromRSAKey: function(pKey, passcode, alg, ivsaltHex) {
410 	    var version = new KJUR.asn1.DERInteger({'int': 0});
411 	    var n = new KJUR.asn1.DERInteger({'bigint': pKey.n});
412 	    var e = new KJUR.asn1.DERInteger({'int': pKey.e});
413 	    var d = new KJUR.asn1.DERInteger({'bigint': pKey.d});
414 	    var p = new KJUR.asn1.DERInteger({'bigint': pKey.p});
415 	    var q = new KJUR.asn1.DERInteger({'bigint': pKey.q});
416 	    var dmp1 = new KJUR.asn1.DERInteger({'bigint': pKey.dmp1});
417 	    var dmq1 = new KJUR.asn1.DERInteger({'bigint': pKey.dmq1});
418 	    var coeff = new KJUR.asn1.DERInteger({'bigint': pKey.coeff});
419 	    var seq = new KJUR.asn1.DERSequence({'array': [version, n, e, d, p, q, dmp1, dmq1, coeff]});
420 	    var hex = seq.getEncodedHex();
421 	    return this.getEryptedPKCS5PEMFromPrvKeyHex(hex, passcode, alg, ivsaltHex);
422         },

424 	/**
425          * generate RSAKey and PEM formatted encrypted PKCS#5 private key
426 	 * @name newEryptedPKCS5PEM
427 	 * @memberOf PKCS5PKEY
428 	 * @function
429 	 * @param {String} passcode pass code to protect private key (ex. password)
430 	 * @param {Integer} keyLen key bit length of RSA key to be generated. (default 1024)
431 	 * @param {String} hPublicExponent hexadecimal string of public exponent (default 10001)
432 	 * @param {String} alg shared key algorithm to encrypt private key (default AES-258-CBC)
433 	 * @return {String} string of PEM formatted encrypted PKCS#5 private key
434          * @since pkcs5pkey 1.0.2
435 	 * @example
436 	 * var pem1 = PKCS5PKEY.newEncryptedPKCS5PEM("password");           // RSA1024bit/10001/AES-256-CBC
437 	 * var pem2 = PKCS5PKEY.newEncryptedPKCS5PEM("password", 512);      // RSA 512bit/10001/AES-256-CBC
438 	 * var pem3 = PKCS5PKEY.newEncryptedPKCS5PEM("password", 512, '3'); // RSA 512bit/    3/AES-256-CBC
439 	 */
440 	newEryptedPKCS5PEM: function(passcode, keyLen, hPublicExponent, alg) {
441 	    if (typeof keyLen == "undefined" || keyLen == null) {
442 		keyLen = 1024;
443 	    }
444 	    if (typeof hPublicExponent == "undefined" || hPublicExponent == null) {
445 		hPublicExponent = '10001';
446 	    }
447 	    var pKey = new RSAKey();
448 	    pKey.generate(keyLen, hPublicExponent);
449 	    var pem = null;
450 	    if (typeof alg == "undefined" || alg == null) {
451 		pem = this.getEncryptedPKCS5PEMFromRSAKey(pkey, passcode);
452 	    } else {
453 		pem = this.getEncryptedPKCS5PEMFromRSAKey(pkey, passcode, alg);
454 	    }
455 	    return pem;
456         },

460 	/**
461          * read PEM formatted unencrypted PKCS#8 private key and returns RSAKey object
462 	 * @name getRSAKeyFromPlainPKCS8PEM
463 	 * @memberOf PKCS5PKEY
464 	 * @function
465 	 * @param {String} pkcs8PEM PEM formatted unencrypted PKCS#8 private key
466 	 * @return {RSAKey} loaded RSAKey object of RSA private key
467          * @since pkcs5pkey 1.0.1
468 	 */
469         getRSAKeyFromPlainPKCS8PEM: function(pkcs8PEM) {
470             if (pkcs8PEM.match(/ENCRYPTED/))
471                 throw "pem shall be not ENCRYPTED";
472 	    if (! pkcs8PEM.match(/BEGIN PRIVATE KEY/))
473                 throw "pkcs8PEM doesn't include 'BEGIN PRIVATE KEY'";
474             var s = pkcs8PEM;
475 	    s = s.replace(/^-----BEGIN PRIVATE KEY-----/, '');
476 	    s = s.replace(/^-----END PRIVATE KEY-----/, '');
477 	    var sB64 = s.replace(/\s+/g, '');
478 	    var prvKeyWA = CryptoJS.enc.Base64.parse(sB64);
479 	    var prvKeyHex = CryptoJS.enc.Hex.stringify(prvKeyWA);
480 	    var a1 = ASN1HEX.getPosArrayOfChildren_AtObj(prvKeyHex, 0);
481 	    if (a1.length != 3)
482 		throw "outer DERSequence shall have 3 elements: " + a1.length;
483             var algIdTLV =ASN1HEX.getHexOfTLV_AtObj(prvKeyHex, a1[1]);
484 	    if (algIdTLV != "300d06092a864886f70d0101010500") // AlgId rsaEncryption
485 		throw "PKCS8 AlgorithmIdentifier is not rsaEnc: " + algIdTLV;
486             var algIdTLV = ASN1HEX.getHexOfTLV_AtObj(prvKeyHex, a1[1]);
487 	    var octetStr = ASN1HEX.getHexOfTLV_AtObj(prvKeyHex, a1[2]);
488 	    var p5KeyHex = ASN1HEX.getHexOfV_AtObj(octetStr, 0);
489             //alert(p5KeyHex);
490 	    var rsaKey = new RSAKey();
491 	    rsaKey.readPrivateKeyFromASN1HexString(p5KeyHex);
492 	    return rsaKey;
