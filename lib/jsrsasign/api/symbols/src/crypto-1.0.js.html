 65     /**
 66      * get hexadecimal DigestInfo
 67      * @name getDigestInfoHex
 68      * @memberOf KJUR.crypto.Util
 69      * @function
 70      * @param {String} hHash hexadecimal hash value
 71      * @param {String} alg hash algorithm name (ex. 'sha1')
 72      * @return {String} hexadecimal string DigestInfo ASN.1 structure
 73      */
 74     this.getDigestInfoHex = function(hHash, alg) {
 75 	if (typeof this.DIGESTINFOHEAD[alg] == "undefined")
 76 	    throw "alg not supported in Util.DIGESTINFOHEAD: " + alg;
 77 	return this.DIGESTINFOHEAD[alg] + hHash;
 78     };

 80     /**
 81      * get PKCS#1 padded hexadecimal DigestInfo
 82      * @name getPaddedDigestInfoHex
 83      * @memberOf KJUR.crypto.Util
 84      * @function
 85      * @param {String} hHash hexadecimal hash value
 86      * @param {String} alg hash algorithm name (ex. 'sha1')
 87      * @param {Integer} keySize key bit length (ex. 1024)
 88      * @return {String} hexadecimal string of PKCS#1 padded DigestInfo
 89      */
 90     this.getPaddedDigestInfoHex = function(hHash, alg, keySize) {
 91 	var hDigestInfo = this.getDigestInfoHex(hHash, alg);
 92 	var pmStrLen = keySize / 4; // minimum PM length
 93 
 94 	if (hDigestInfo.length + 22 > pmStrLen) // len(0001+ff(*8)+00+hDigestInfo)=22
 95 	    throw "key is too short for SigAlg: keylen=" + keySize + "," + alg;
 96 
 97 	var hHead = "0001";
 98 	var hTail = "00" + hDigestInfo;
 99 	var hMid = "";
100 	var fLen = pmStrLen - hHead.length - hTail.length;
101 	for (var i = 0; i < fLen; i += 2) {
102 	    hMid += "ff";
103 	}
104 	var hPaddedMessage = hHead + hMid + hTail;
105 	return hPaddedMessage;
106     };

179 /**
180  * MessageDigest class which is very similar to java.security.MessageDigest class
181  * @name KJUR.crypto.MessageDigest
182  * @class MessageDigest class which is very similar to java.security.MessageDigest class
183  * @param {Array} params parameters for constructor
184  * @description
185  * <br/>
186  * Currently this supports md5, sha1, sha224, sha256, sha384,
187  * sha512 and ripemd160 for algorithm and 'cryptojs' for provider.
188  * @example
189  * var md = new KJUR.crypto.MessageDigest({"alg": "sha1", "prov": "cryptojs"});
190  * // append data
191  * md.updateHex('1f2d3e')
192  * md.updateString('aaa')
193  * // get message digest
194  * md.digest()
195  * md.digestHex('5f6de0')
196  * md.digestString('aaa')
197  * @see http://crypto-js.googlecode.com/svn/tags/3.1.2/build/components/core-min.js
198  * @see http://crypto-js.googlecode.com/svn/tags/3.1.2/build/rollups/sha1.js
199  */
200 KJUR.crypto.MessageDigest = function(params) {
201     var md = null;
202     var algName = null;
203     var provName = null;
204     var _CryptoJSMdName = {
205 	'md5': 'CryptoJS.algo.MD5',
206 	'sha1': 'CryptoJS.algo.SHA1',
207 	'sha224': 'CryptoJS.algo.SHA224',
208 	'sha256': 'CryptoJS.algo.SHA256',
209 	'sha384': 'CryptoJS.algo.SHA384',
210 	'sha512': 'CryptoJS.algo.SHA512',
211 	'ripemd160': 'CryptoJS.algo.RIPEMD160'
212     };
213 
214     /**
215      * set hash algorithm and provider
216      * @name setAlgAndProvider
217      * @memberOf KJUR.crypto.MessageDigest
218      * @function
219      * @param {String} alg hash algorithm name
220      * @param {String} prov provider name
221      * @description
222      * @example
223      * // for SHA1
224      * md.setAlgAndProvider('sha1', 'cryptojs');
225      * // for RIPEMD160
226      * md.setAlgAndProvider('ripemd160', 'cryptojs');
227      */
228     this.setAlgAndProvider = function(alg, prov) {
229 	if (':md5:sha1:sha224:sha256:sha384:sha512:ripemd160:'.indexOf(alg) != -1 &&
230 	    prov == 'cryptojs') {
231 	    try {
232 		this.md = eval(_CryptoJSMdName[alg]).create();
233 	    } catch (ex) {
234 		throw "setAlgAndProvider hash alg set fail alg=" + alg + "/" + ex;
235 	    }
236 	    this.updateString = function(str) {
237 		this.md.update(str);
238 	    };
239 	    this.updateHex = function(hex) {
240 		var wHex = CryptoJS.enc.Hex.parse(hex);
241 		this.md.update(wHex);
242 	    };
243 	    this.digest = function() {
244 		var hash = this.md.finalize();
245 		return hash.toString(CryptoJS.enc.Hex);
246 	    };
247 	    this.digestString = function(str) {
248 		this.updateString(str);
249 		return this.digest();
250 	    };
251 	    this.digestHex = function(hex) {
252 		this.updateHex(hex);
253 		return this.digest();
254 	    };
255 	}
256     };

337 /**
338  * Signature class which is very similar to java.security.Signature class
339  * @name KJUR.crypto.Signature
340  * @class Signature class which is very similar to java.security.Signature class
341  * @param {Array} params parameters for constructor
342  * @property {String} state Current state of this signature object whether 'SIGN', 'VERIFY' or null
343  * @description
344  * <br/>
345  * As for params of constructor's argument, it can be specify following attributes:
346  * <ul>
347  * <li>alg - signature algorithm name (ex. {MD5,SHA1,SHA224,SHA256,SHA384,SHA512,RIPEMD160}withRSA)</li>
348  * <li>provider - currently 'cryptojs/jsrsa' only</li>
349  * <li>prvkeypem - PEM string of signer's private key. If this specified, no need to call initSign(prvKey).</li>
350  * </ul>
351  * <h4>SUPPORTED ALGORITHMS AND PROVIDERS</h4>
352  * Signature class supports {MD5,SHA1,SHA224,SHA256,SHA384,SHA512,RIPEMD160}
353  * withRSA algorithm in 'cryptojs/jsrsa' provider.
354  * <h4>EXAMPLES</h4>
355  * @example
356  * // signature generation
357  * var sig = new KJUR.crypto.Signature({"alg": "SHA1withRSA", "prov": "cryptojs/jsrsa"});
358  * sig.initSign(prvKey);
359  * sig.updateString('aaa');
360  * var hSigVal = sig.sign();
361  *
362  * // signature validation
363  * var sig2 = new KJUR.crypto.Signature({"alg": "SHA1withRSA", "prov": "cryptojs/jsrsa"});
364  * sig2.initVerifyByCertificatePEM(cert)
365  * sig.updateString('aaa');
366  * var isValid = sig2.verify(hSigVal);
367  */
368 KJUR.crypto.Signature = function(params) {
369     var prvKey = null; // RSAKey for signing
370     var pubKey = null; // RSAKey for verifying
371 
372     var md = null; // KJUR.crypto.MessageDigest object
373     var sig = null;
374     var algName = null;
375     var provName = null;
376     var algProvName = null;
377     var mdAlgName = null;
378     var pubkeyAlgName = null;
379     var state = null;
380 
381     var sHashHex = null; // hex hash value for hex
382     var hDigestInfo = null;
383     var hPaddedDigestInfo = null;
384     var hSign = null;
385 
386     this._setAlgNames = function() {
387 	if (this.algName.match(/^(.+)with(.+)$/)) {
388 	    this.mdAlgName = RegExp.$1.toLowerCase();
389 	    this.pubkeyAlgName = RegExp.$2.toLowerCase();
390 	}
391     };
392 
393     this._zeroPaddingOfSignature = function(hex, bitLength) {
394 	var s = "";
395 	var nZero = bitLength / 4 - hex.length;
396 	for (var i = 0; i < nZero; i++) {
397 	    s = s + "0";
398 	}
399 	return s + hex;
400     };
401 
402     /**
403      * set signature algorithm and provider
404      * @name setAlgAndProvider
405      * @memberOf KJUR.crypto.Signature
406      * @function
407      * @param {String} alg signature algorithm name
408      * @param {String} prov provider name
409      * @description
410      * @example
411      * md.setAlgAndProvider('SHA1withRSA', 'cryptojs/jsrsa');
412      */
413     this.setAlgAndProvider = function(alg, prov) {
414 	this._setAlgNames();
415 	if (prov != 'cryptojs/jsrsa')
416 	    throw "provider not supported: " + prov;
417 
418 	if (':md5:sha1:sha224:sha256:sha384:sha512:ripemd160:'.indexOf(this.mdAlgName) != -1) {
419 	    try {
420 		this.md = new KJUR.crypto.MessageDigest({'alg':this.mdAlgName,'prov':'cryptojs'});
421 	    } catch (ex) {
422 		throw "setAlgAndProvider hash alg set fail alg=" + this.mdAlgName + "/" + ex;
423 	    }
424 
425 	    this.initSign = function(prvKey) {
426 		this.prvKey = prvKey;
427 		this.state = "SIGN";
428 	    };
429 
430 	    this.initVerifyByPublicKey = function(rsaPubKey) {
431 		this.pubKey = rsaPubKey;
432 		this.state = "VERIFY";
433 	    };
434 
435 	    this.initVerifyByCertificatePEM = function(certPEM) {
436 		var x509 = new X509();
437 		x509.readCertPEM(certPEM);
438 		this.pubKey = x509.subjectPublicKeyRSA;
439 		this.state = "VERIFY";
440 	    };
441 
442 	    this.updateString = function(str) {
443 		this.md.updateString(str);
444 	    };
445 	    this.updateHex = function(hex) {
446 		this.md.updateHex(hex);
447 	    };
448 	    this.sign = function() {
449                 var util = KJUR.crypto.Util;
450 		var keyLen = this.prvKey.n.bitLength();
451 		this.sHashHex = this.md.digest();
452 		this.hDigestInfo = util.getDigestInfoHex(this.sHashHex, this.mdAlgName);
453 		this.hPaddedDigestInfo = 
454                     util.getPaddedDigestInfoHex(this.sHashHex, this.mdAlgName, keyLen);
455 
456 		var biPaddedDigestInfo = parseBigInt(this.hPaddedDigestInfo, 16);
457 		this.hoge = biPaddedDigestInfo.toString(16);
458 
459 		var biSign = this.prvKey.doPrivate(biPaddedDigestInfo);
460 		this.hSign = this._zeroPaddingOfSignature(biSign.toString(16), keyLen);
461 		return this.hSign;
462 	    };
463 	    this.signString = function(str) {
464 		this.updateString(str);
465 		this.sign();
466 	    };
467 	    this.signHex = function(hex) {
468 		this.updateHex(hex);
469 		this.sign();
470 	    };
471 	    this.verify = function(hSigVal) {
472                 var util = KJUR.crypto.Util;
473 		var keyLen = this.pubKey.n.bitLength();
474 		this.sHashHex = this.md.digest();
475 
476 		var biSigVal = parseBigInt(hSigVal, 16);
477 		var biPaddedDigestInfo = this.pubKey.doPublic(biSigVal);
478 		this.hPaddedDigestInfo = biPaddedDigestInfo.toString(16);
479                 var s = this.hPaddedDigestInfo;
480                 s = s.replace(/^1ff+00/, '');
481 
482 		var hDIHEAD = KJUR.crypto.Util.DIGESTINFOHEAD[this.mdAlgName];
483                 if (s.indexOf(hDIHEAD) != 0) {
484 		    return false;
485 		}
486 		var hHashFromDI = s.substr(hDIHEAD.length);
487 		//alert(hHashFromDI + "\n" + this.sHashHex);
488 		return (hHashFromDI == this.sHashHex);
489 	    };
490 	}
491     };

609     /**
610      * verifies the passed-in signature.
611      * @name verify
612      * @memberOf KJUR.crypto.Signature
613      * @function
614      * @param {String} str string to final update
615      * @return {Boolean} true if the signature was verified, otherwise false
616      * @description
617      * @example
618      * var isValid = sig.verify('1fbcefdca4823a7(snip)')
619      */
620     this.verify = function(hSigVal) {
621 	throw "verify(hSigVal) not supported for this alg:prov=" + this.algProvName;
622     };
623 
624     if (typeof params != "undefined") {
625 	if (typeof params['alg'] != "undefined") {
626 	    this.algName = params['alg'];
627 	    this.provName = params['prov'];
628 	    this.algProvName = params['alg'] + ":" + params['prov'];
629 	    this.setAlgAndProvider(params['alg'], params['prov']);
630 	    this._setAlgNames();
631 	}
632 	if (typeof params['prvkeypem'] != "undefined") {
633 	    if (typeof params['prvkeypas'] != "undefined") {
634 		throw "both prvkeypem and prvkeypas parameters not supported";
635 	    } else {
636 		try {
637 		    var prvKey = new RSAKey();
638 		    prvKey.readPrivateKeyFromPEMString(params['prvkeypem']);
639 		    this.initSign(prvKey);
640 		} catch (ex) {
641 		    throw "fatal error to load pem private key: " + ex;
642 		}
643 	    }
644 	}
645     }
